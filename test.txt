-- Получение здоровья и максимального здоровья игрока из всех возможных систем
local function getPlayerHealth(player)
    local char = player.Character
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    
    local health = hum and hum.Health or 0
    local maxHealth = hum and hum.MaxHealth or 100
    
    if char then
        -- Поиск атрибутов
        if char:GetAttribute("Health") then
            health = char:GetAttribute("Health")
            maxHealth = char:GetAttribute("MaxHealth") or 100
        elseif player:GetAttribute("Health") then
            health = player:GetAttribute("Health")
            maxHealth = player:GetAttribute("MaxHealth") or 100
        end
        -- Поиск Value объектов (для кастомных шутеров)
        local healthVal = char:FindFirstChild("Health") or char:FindFirstChild("health")
        if healthVal and healthVal:IsA("ValueBase") then
            health = healthVal.Value
            local maxHealthVal = char:FindFirstChild("MaxHealth") or char:FindFirstChild("maxHealth")
            if maxHealthVal and maxHealthVal:IsA("ValueBase") then
                maxHealth = maxHealthVal.Value
            end
        end
    end
    
    if maxHealth <= 0 then maxHealth = 100 end
    return health, maxHealth
end

-- Проверка видимости через Raycast
local function isVisible(part, character)
    local origin = camera.CFrame.Position
    local direction = part.Position - origin
    local raycastParams = RaycastParams.new()
    raycastParams.FilterType = Enum.RaycastFilterType.Exclude
    raycastParams.FilterDescendantsInstances = {localPlayer.Character, character}
    local result = workspace:Raycast(origin, direction, raycastParams)
    return result == nil
end

-- Поиск ближайшей цели с учетом проверок на команду и стены
local function getClosestPlayer()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local mousePos = game:GetService("UserInputService"):GetMouseLocation()

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local health, maxHealth = getPlayerHealth(player)
            if health > 0 then
                -- Team Check
                if teamCheckEnabled and player.Team == localPlayer.Team then
                    continue
                end
                
                local rootPart = player.Character:FindFirstChild(targetPart) or player.Character.HumanoidRootPart
                
                -- Visible Check
                if visibleCheckEnabled and not isVisible(rootPart, player.Character) then
                    continue
                end

                local screenPos, onScreen = camera:WorldToViewportPoint(rootPart.Position)
                if onScreen then
                    local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                    if distance < shortestDistance and distance <= fovCircle.Radius then
                        shortestDistance = distance
                        closestPlayer = player
                    end
                end
            end
        end
    end
    return closestPlayer
end

-- === METATABLE HOOK FOR SILENT AIM ===
task.spawn(function()
    pcall(function()
        local mt = getrawmetatable(game)
        local oldIndex = mt.__index
        local oldNamecall = mt.__namecall
        setreadonly(mt, false)
        
        -- Хук __index для перенаправления Silent Aim (Mouse.Hit / Camera)
        mt.__index = newcclosure(function(self, key)
            if not checkcaller() and silentAimEnabled and key == "Hit" then
                local targetPlayer = getClosestPlayer()
                if targetPlayer and targetPlayer.Character then
                    local part = targetPlayer.Character:FindFirstChild(targetPart) or targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                    if part then
                        local finalPos = part.Position
                        if predictionEnabled then
                            finalPos = finalPos + (part.AssemblyLinearVelocity * predictionAmount)
                        end
                        return CFrame.new(finalPos)
                    end
                end
            end
            return oldIndex(self, key)
        end)
        
        -- Хук __namecall для Silent Aim трассировки пули
        mt.__namecall = newcclosure(function(self, ...)
            local method = getnamecallmethod()
            local args = {...}
            
            if not checkcaller() then
                if silentAimEnabled and (method == "Raycast" or method == "raycast") and self == workspace then
                    local targetPlayer = getClosestPlayer()
                    if targetPlayer and targetPlayer.Character then
                        local part = targetPlayer.Character:FindFirstChild(targetPart) or targetPlayer.Character:FindFirstChild("HumanoidRootPart")
                        if part then
                            local origin = args[1]
                            local spoofRayDirection = (part.Position - origin).Unit * (part.Position - origin).Magnitude
                            args[2] = spoofRayDirection
                            return oldNamecall(self, unpack(args))
                        end
                    end
                end
            end
            return oldNamecall(self, ...)
        end)
        
        setreadonly(mt, true)
    end)
end)

local function formatUptime(seconds)
    local hours = math.floor(seconds / 3600)
    local minutes = math.floor((seconds % 3600) / 60)
    local secs = math.floor(seconds % 60)
    return string.format("%dh %dm %ds", hours, minutes, secs)

local xrayEnabled = false
local xrayTransparency = 0.5
local PlayersService = game:GetService("Players")

local function applyXray(instance)
    if not instance:IsA("BasePart") then return end
    
    local ancestor = instance.Parent
    local isPlayerPart = false
    while ancestor and ancestor ~= workspace do
        if ancestor:IsA("Model") and PlayersService:GetPlayerFromCharacter(ancestor) then
            isPlayerPart = true
            break
        end
        ancestor = ancestor.Parent
    end
    if isPlayerPart then return end
    
    if xrayEnabled then
        if instance.Transparency >= 0.9 then return end
        if not originalTransparencies[instance] then
            originalTransparencies[instance] = instance.Transparency
        end
        instance.Transparency = xrayTransparency
    else
        if originalTransparencies[instance] then
            instance.Transparency = originalTransparencies[instance]
            originalTransparencies[instance] = nil
        end
    end
end

VisualsSection:AddToggle({
    Name = "Xray",
    Value = false,
    Flag = "XrayEnabled",
    Callback = function(val)
        xrayEnabled = val
        if currentXrayThread then task.cancel(currentXrayThread) currentXrayThread = nil end
        
        currentXrayThread = task.spawn(function()
            if xrayEnabled then
                local descendants = workspace:GetDescendants()
                local count = 0
                for i = 1, #descendants do
                    if not xrayEnabled then break end
                    local desc = descendants[i]
                    applyXray(desc)
                    count = count + 1
                    if count >= 150 then count = 0 task.wait() end
                end
            else
                local count = 0
                for instance, origVal in pairs(originalTransparencies) do
                    if xrayEnabled then break end
                    if instance and instance.Parent then instance.Transparency = origVal end
                    originalTransparencies[instance] = nil
                    count = count + 1
                    if count >= 150 then count = 0 task.wait() end
                end
                table.clear(originalTransparencies)
            end
            currentXrayThread = nil
        end)
    end
})

                    if math.abs(value - 14) > 0.01 then origClockTime = value end
                elseif prop == "GlobalShadows" then
                    local value = Lighting.GlobalShadows
                    if value ~= false then origGlobalShadows = value end
                end
                
                Lighting.Ambient = Color3.fromRGB(255, 255, 255)
                Lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
                Lighting.Brightness = 2
                Lighting.ClockTime = 14
                Lighting.GlobalShadows = false
            end)

            Lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
            Lighting.Brightness = 2
            Lighting.ClockTime = 14
            Lighting.GlobalShadows = false
        else
            if fullbrightConnection then fullbrightConnection:Disconnect() fullbrightConnection = nil end
            Lighting.Ambient = origAmbient
            Lighting.OutdoorAmbient = origOutdoorAmbient
            Lighting.Brightness = origBrightness

                local root = player.Character:FindFirstChild("HumanoidRootPart")

                        if not originalHitboxSizes[player] then
                            originalHitboxSizes[player] = root.Size
                        end
                        root.Size = Vector3.new(hitboxSize, hitboxSize, hitboxSize)
                        root.Transparency = 0.7
                        root.CanCollide = true
                    else
                        if originalHitboxSizes[player] then
                            root.Size = originalHitboxSizes[player]

                            originalHitboxSizes[player] = nil
                        end
                    end
                end
            end
        end
    end
end)

-- Цикл Aimbot и FOV Circle
aimbotConnection = RunService.RenderStepped:Connect(function()
    if not library.IsGuiValid() then
        if aimbotConnection then aimbotConnection:Disconnect() end
        return
    end
    
    -- Обновление позиции, радиуса и цвета FOV
    if showFov then
        local mouseLocation = game:GetService("UserInputService"):GetMouseLocation()
        fovCircle.Position = mouseLocation
        fovCircle.Radius = fovRadius
        fovCircle.Color = fovColor
        fovCircle.Visible = true
    else
        fovCircle.Visible = false
    end

    -- Наведение аима
    if aimbotEnabled then
        local UIS = game:GetService("UserInputService")
        local isPressed = false
        if aimbotKey.Name:find("MouseButton") then
            isPressed = UIS:IsMouseButtonPressed(aimbotKey)
        else
            isPressed = UIS:IsKeyDown(aimbotKey)
        end
        
        if isPressed then
            local target = getClosestPlayer()
            if target and target.Character then
                local part = target.Character:FindFirstChild(targetPart)
                if part then
                    local targetPos = part.Position
                    if predictionEnabled then
                        targetPos = targetPos + (part.AssemblyLinearVelocity * predictionAmount)
                    end
                    camera.CFrame = camera.CFrame:Lerp(CFrame.new(camera.CFrame.Position, targetPos), aimbotSmoothness)
                end
            end
        end
    end
end)

-- Оптимизированный и быстрый цикл Triggerbot на основе Raycast
triggerbotConnection = RunService.RenderStepped:Connect(function()
    if not library.IsGuiValid() then
        if triggerbotConnection then triggerbotConnection:Disconnect() end
        return
    end
    if triggerbotEnabled then
        local mousePos = game:GetService("UserInputService"):GetMouseLocation()
        local ray = camera:ViewportPointToRay(mousePos.X, mousePos.Y)
        
        local raycastParams = RaycastParams.new()
        raycastParams.FilterType = Enum.RaycastFilterType.Exclude
        raycastParams.FilterDescendantsInstances = {localPlayer.Character}
        
        local result = workspace:Raycast(ray.Origin, ray.Direction * 2000, raycastParams)
        if result and result.Instance then
            local hitPart = result.Instance
            local model = hitPart.Parent
            if model:IsA("Accessory") then model = model.Parent end
            
            local enemyPlayer = Players:GetPlayerFromCharacter(model)
            if enemyPlayer and enemyPlayer ~= localPlayer then
                if teamCheckEnabled and enemyPlayer.Team == localPlayer.Team then
                    return
                end
                local health, maxHealth = getPlayerHealth(enemyPlayer)
                if health > 0 then
                    task.wait(triggerbotDelay)
                    if mouse1click then
                        mouse1click()
                    elseif mouse1press and mouse1release then
                        mouse1press()
                        task.wait(0.01)
                        mouse1release()
                    end
                end
            end
        end
    end
end)

-- Цикл ESP и прорисовки линий (Bones/Boxes/Tracers/Health)
espConnection = RunService.RenderStepped:Connect(function()
    if not library.IsGuiValid() or not espEnabled then
        -- Скрываем все элементы, если ESP выключен
        for _, container in pairs(espDrawings) do
            container.Box.Visible = false
            container.Name.Visible = false
            container.Distance.Visible = false
            container.Tracer.Visible = false
            container.HealthBarOutline.Visible = false
            container.HealthBar.Visible = false
            for _, bone in pairs(container.Bones) do
                bone.Visible = false
            end
        end
        if not library.IsGuiValid() then
            if espConnection then espConnection:Disconnect() end
            for player in pairs(espDrawings) do
                removeEsp(player)
            end
        end
        return
    end

    -- Определение точки исхода трейсеров
    local screenOrigin = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y) -- По дефолту Bottom
    if tracersOrigin == "Center" then
        screenOrigin = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)
    elseif tracersOrigin == "Mouse" then
        screenOrigin = game:GetService("UserInputService"):GetMouseLocation()
    end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= localPlayer then
            createEsp(player)
            local container = espDrawings[player]
            local char = player.Character
            local root = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            
            -- Проверка на команду перед отрисовкой
            if root and hum then
                local health, maxHealth = getPlayerHealth(player)
                
                -- Если здоровье игрока <= 0, мгновенно скрываем его ESP чертежи во избежание фантомных рамок
                if health <= 0 then
                    container.Box.Visible = false
                    container.Name.Visible = false
                    container.Distance.Visible = false
                    container.Tracer.Visible = false
                    container.HealthBarOutline.Visible = false
                    container.HealthBar.Visible = false
                    for _, bone in pairs(container.Bones) do bone.Visible = false end
                    continue
                end

                if teamCheckEnabled and player.Team == localPlayer.Team then
                    container.Box.Visible = false
                    container.Name.Visible = false
                    container.Distance.Visible = false
                    container.Tracer.Visible = false
                    container.HealthBarOutline.Visible = false
                    container.HealthBar.Visible = false
                    for _, bone in pairs(container.Bones) do bone.Visible = false end
                    continue
                end

                local rootPos, onScreen = camera:WorldToViewportPoint(root.Position)
                
                if onScreen then
                    local head = char:FindFirstChild("Head")
                    local boxHeight, boxWidth = 0, 0
                    local headPos
                    
                    if head then
                        headPos = camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                        local legPos = camera:WorldToViewportPoint(root.Position - Vector3.new(0, 3, 0))
                        boxHeight = math.abs(headPos.Y - legPos.Y)
                        boxWidth = boxHeight * 0.6
                    end

                    -- 2D Boxes
                    if showEspBox and head then
                        container.Box.Size = Vector2.new(boxWidth, boxHeight)
                        container.Box.Position = Vector2.new(rootPos.X - boxWidth/2, rootPos.Y - boxHeight/2)
                        container.Box.Color = espColor
                        container.Box.Visible = true
                    else
                        container.Box.Visible = false
                    end
                    
                    -- Health Bar (Индикатор здоровья с защитой от багов и плавным градиентом)
                    if showHealthBar and head then
                        local barX = rootPos.X - boxWidth/2 - 6
                        local barY_bottom = rootPos.Y + boxHeight/2
                        local barY_top = rootPos.Y - boxHeight/2
                        
                        container.HealthBarOutline.From = Vector2.new(barX, barY_bottom + 1)
                        container.HealthBarOutline.To = Vector2.new(barX, barY_top - 1)
                        container.HealthBarOutline.Visible = true
                        
                        local currentHealth = math.clamp(health, 0, maxHealth)
                        local healthPercent = currentHealth / maxHealth
                        
                        local healthY = barY_bottom - (boxHeight * healthPercent)
                        
                        container.HealthBar.From = Vector2.new(barX, barY_bottom)
                        container.HealthBar.To = Vector2.new(barX, healthY)
                        
                        -- Плавное изменение цвета от зеленого (120) к красному (0) по HSV
                        container.HealthBar.Color = Color3.fromHSV((healthPercent * 120) / 360, 1, 1)
                        container.HealthBar.Visible = true
                    else
                        container.HealthBarOutline.Visible = false
                        container.HealthBar.Visible = false
                    end

                    -- Names
                    if showEspNames then
                        container.Name.Text = player.DisplayName or player.Name
                        container.Name.Position = Vector2.new(rootPos.X, rootPos.Y - (boxHeight / 2) - 15)
                        container.Name.Visible = true
                    else
                        container.Name.Visible = false
                    end
                    
                    -- Distance
                    if showEspDistance then
                        local dist = math.round((root.Position - camera.CFrame.Position).Magnitude)
                        container.Distance.Text = tostring(dist) .. " studs"
                        container.Distance.Position = Vector2.new(rootPos.X, rootPos.Y + (boxHeight / 2) + 5)
                        container.Distance.Visible = true
                    else
                        container.Distance.Visible = false
                    end

                    -- Tracers
                    if showTracers then
                        container.Tracer.From = screenOrigin
                        container.Tracer.To = Vector2.new(rootPos.X, rootPos.Y)
                        container.Tracer.Color = espColor
                        container.Tracer.Visible = true
                    else
                        container.Tracer.Visible = false
                    end
                    
                    -- Skeletons
                    if showEspSkeleton then
                        local upperTorso = char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
                        local lowerTorso = char:FindFirstChild("LowerTorso") or char:FindFirstChild("Torso")
                        local leftArm = char:FindFirstChild("LeftHand") or char:FindFirstChild("Left Arm")
                        local rightArm = char:FindFirstChild("RightHand") or char:FindFirstChild("Right Arm")
                        local leftLeg = char:FindFirstChild("LeftFoot") or char:FindFirstChild("Left Leg")
                        local rightLeg = char:FindFirstChild("RightFoot") or char:FindFirstChild("Right Leg")
                        
                        local function drawBone(line, p1, p2)
                            if p1 and p2 then
                                local pos1, os1 = camera:WorldToViewportPoint(p1.Position)
                                local pos2, os2 = camera:WorldToViewportPoint(p2.Position)
                                if os1 and os2 then
                                    line.From = Vector2.new(pos1.X, pos1.Y)
                                    line.To = Vector2.new(pos2.X, pos2.Y)
                                    line.Color = espColor
                                    line.Visible = true
                                    return
                                end
                            end
                            line.Visible = false
                        end
                        
                        drawBone(container.Bones.HeadTorso, head, upperTorso)
                        drawBone(container.Bones.Spine, upperTorso, lowerTorso)
                        drawBone(container.Bones.LeftArm, upperTorso, leftArm)
                        drawBone(container.Bones.RightArm, upperTorso, rightArm)
                        drawBone(container.Bones.LeftLeg, lowerTorso, leftLeg)
                        drawBone(container.Bones.RightLeg, lowerTorso, rightLeg)
                    else
                        for _, bone in pairs(container.Bones) do
                            bone.Visible = false
                        end
                    end
                else
                    container.Box.Visible = false
                    container.Name.Visible = false
                    container.Distance.Visible = false
                    container.Tracer.Visible = false
                    container.HealthBarOutline.Visible = false
                    container.HealthBar.Visible = false
                    for _, bone in pairs(container.Bones) do
                        bone.Visible = false
                    end
                end
            else
                container.Box.Visible = false
                container.Name.Visible = false
                container.Distance.Visible = false
                container.Tracer.Visible = false
                container.HealthBarOutline.Visible = false
                container.HealthBar.Visible = false
                for _, bone in pairs(container.Bones) do
                    bone.Visible = false
                end
            end
        end
    end
    
    -- Очистка неактивных игроков
    for player in pairs(espDrawings) do
        if not Players:FindFirstChild(player.Name) then
            removeEsp(player)
        end
    end
end)

local Stats = game:GetService("Stats")

local frameCount = 0
local lastTime = os.clock()
local currentFps = 60

local fpsConnection
fpsConnection = RunService.RenderStepped:Connect(function()
    if not library.IsGuiValid() then
        fpsConnection:Disconnect()
        return
    end
    frameCount = frameCount + 1
end)

task.spawn(function()
    local Players = game:GetService("Players")
    while Wait(0.5) do
        local currentTime = os.clock()
        local elapsed = currentTime - lastTime
        
        if elapsed > 0 then
            currentFps = math.round(frameCount / elapsed)
        end
        
        frameCount = 0
        lastTime = currentTime

        local ping = 0
        pcall(function()
            local lp = Players.LocalPlayer
            if lp then ping = math.round(lp:GetNetworkPing() * 1000) end
        end)
        
        if ping <= 0 then
            pcall(function()
                ping = math.round(Stats.Network.ServerStatsItem["Data Ping"]:GetValue())
            end)
        end
        
        pingLabel:Set("Ping: " .. tostring(ping) .. " ms")
        fpsLabel:Set("FPS: " .. tostring(currentFps))
        timeLabel:Set("Local Time: " .. os.date("%X"))
    end
end)
